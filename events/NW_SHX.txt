##Вооружённый конфликт между армиями
country_event = {
	id = shanxi.10
	title = shanxi.10.t
	desc = shanxi.10.desc
	is_triggered_only = yes
	immediate = { log = "[GetDateText]: [Root.GetName] ([Root.GetTag]): event shanxi.10" }

	option = {
		name = shanxi.10.a
		remove_power_balance = {
			id = SHX_old_and_new_army_balance
		}
		start_civil_war = {
			ideology = communism
			size = 0.5
			
			PREV = {
				SHX_bo_yibo = {
					set_nationality = PREV.PREV
					promote_character = yes
				}
				SHX_xu_fanting = {
					set_nationality = PREV.PREV
				}
			}
			complete_national_focus = SHX_purges_of_the_old_army
		}
		complete_national_focus = SHX_create_an_iron_army
	}
	option = {
		name = shanxi.10.b
		remove_power_balance = {
			id = SHX_old_and_new_army_balance
		}
		start_civil_war = {
			ideology = neutrality
			ruling_party = communism
			size = 0.5
			
			complete_national_focus = SHX_create_an_iron_army
		}
		promote_character = SHX_bo_yibo
		complete_national_focus = SHX_purges_of_the_old_army
	}
}
