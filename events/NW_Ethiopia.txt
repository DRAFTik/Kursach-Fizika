country_event = {
	id = NWethiopia.4
	title = NWethiopia.4.t
	desc = NWethiopia.4.desc
	picture = GFX_report_event_generic_rally_3
	is_triggered_only = yes
	
	immediate = { log = "[GetDateText]: [Root.GetName] ([Root.GetTag]): event NWethiopia.4" }
	
	option = {
		name = NWethiopia.4.a
		ai_chance = {
			factor = 40
		}
		every_other_country = {
			limit = {
				OR = {
					has_country_flag = support_league_nations_gases_flag
					tag = ENG
				}
			}
			add_opinion_modifier = { target = ROOT modifier = ETH_use_of_gas }
			ROOT = { add_opinion_modifier = { target = PREV modifier = ETH_use_of_gas } }
		}
		ENG = { country_event = { id = NWethiopia.5 days = 2 } }
	}
	
	option = {
		name = NWethiopia.4.b
		ai_chance = {
			factor = 60
			modifier = {
				factor = 0
				OR = {
					is_historical_focus_on = yes
					has_country_flag = NW_ITA_AI_FASCIST_HISTORICAL
				}
			}
		}
		if = {
			limit = { has_idea = ITA_gas_idea }
			remove_ideas = ITA_gas_idea
		}
	}
}

#Вооружение Эфиопов // FRA / ENG
country_event = {
	id = NWethiopia.8
	title = NWethiopia.8.t
	desc = NWethiopia.8.desc
	picture = GFX_report_event_generic_rally_3
	is_triggered_only = yes
	
	immediate = { log = "[GetDateText]: [Root.GetName] ([Root.GetTag]): event NWethiopia.8" }
	
	option = {
		name = NWethiopia.8.a
		ai_chance = {
			factor = 90
			modifier = {
				factor = 0
				OR = {
					has_civil_war = yes
					has_war_with = FRC
					has_war_with = FRF
				}
			}
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}
		}
		add_equipment_to_stockpile = {
			type = infantry_equipment
			amount = -1500
		}
		FROM = {
			if = {
				limit = { original_tag = ENG }
				add_equipment_to_stockpile = {
					type = infantry_equipment
					amount = 1500
					producer = ENG
				}
			}
			if = {
				limit = { original_tag = FRA }
				add_equipment_to_stockpile = {
					type = infantry_equipment
					amount = 1500
					producer = FRA
				}
			}
		}
		add_opinion_modifier = { target = FROM modifier = ETH_help_in_war }
		FROM = { add_opinion_modifier = { target = ROOT modifier = ETH_help_in_war } }
	}
	
	option = {
		name = NWethiopia.8.b
		ai_chance = {
			factor = 10
		}
	}
}
