	is_triggered_only = yes

	immediate = { log = "[GetDateText]: [Root.GetName]: event NWbelgium.5" }
	
	option = {
		name = NWbelgium.5.a
		ai_chance = {
			factor = 90
		}
		if = {
			limit = { ROOT = { is_faction_leader = yes } }
			add_to_faction = FROM
		}
		if = {
			limit = {
				ROOT = {
					is_in_faction = yes
					is_faction_leader = no
				}
			}
			random_country = {
				limit = {
					is_in_faction_with = ROOT
					is_faction_leader = yes
				}
				add_to_faction = FROM
			}
		}
		if = {
			limit = {
				ROOT = {
					original_tag = event_target:NW_france_scope
					is_in_faction = no
				}
			}
			set_rule = { can_create_factions = yes }
			create_faction = french_entente_alliance
			add_to_faction = FROM
			hidden_effect = {
				news_event = { hours = 6 id = news.177 }
			}
		}
		if = {
			limit = {
				ROOT = {
					original_tag = FRC
					is_in_faction = no
				}
			}
			set_rule = { can_create_factions = yes }
			create_faction = FRC_french_revolutionary_union
			add_to_faction = FROM
		}
	}
	
	option = {
		name = NWbelgium.5.b
		ai_chance = {
			factor = 10
		}
	}
