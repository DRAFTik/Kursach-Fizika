	id = monaco.2
	title = monaco.2.t
	desc = monaco.2.d
	picture = GFX_report_event_fra_event_1
	trigger = { 
		original_tag = FRA
		MNC = { 
			NOT = { has_government = monarchy } 
			OR = {
				is_subject = no
				is_subject_of = FRA
			}
		} 
		FRA = { 
			exists = yes
			is_subject = no
			NOT = { 
				OR = {
					has_government = trozkism 
					has_government = anarchism
					has_government = communism
				}
			} 
		} 
	}
	
	fire_only_once = yes
	
	option = {
		name = monaco.2.a
		MNC = { country_event = { id = monaco.1 days = 1 } }
	}
	option = {
		name = monaco.2.b
		add_political_power = 10
		MNC = { 
			add_opinion_modifier = { target = FRA modifier = mnc_france35 } 
			remove_ideas = MNC_paris_theatry
		}
		
	}
}
