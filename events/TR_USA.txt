country_event = {

	title = amtrocivwar.9.t
	desc = amtrocivwar.9.d
	picture = GFX_report_event_spain_civil_war_soldiers

	fire_only_once = yes

	trigger = {
		tag = USA
		has_global_flag = usagoinmex_civil_war
		has_global_flag = usa_take_cw
	}
	
	mean_time_to_happen = { days = 90 }
	
	#Гражданка США ТРО
	option = {
		ai_chance = {
			base = 99
		}
		name = amtrocivwar.9.a
		
		hidden_effect = {

			every_country = {
                limit = {
					has_war_with = ROOT
				}
                white_peace = ROOT
			}
			
			every_country = {
                limit = {
					has_war_with = ETH
					NOT = { tag = ITA }
				}
                white_peace = ETH
			}
			
			USA = {
				start_civil_war = {
					ideology = trozkism
					size = 0.4
					capital = 391
					states = { 386 385 378 387 379 377 380 388 381 382 376 389 390 384 383 374 375 391 394 392 373 372 }
				}
				
				set_politics = {
					ruling_party = democratic
					elections_allowed = yes
				}
				add_popularity = {
					ideology = democratic
					popularity = 0.5
				}
				add_popularity = {
					ideology = trozkism
					popularity = -1
				}
				set_stability = 0.2
				news_event = { hours = 4 id = trworldnews.13 }
				set_global_flag = usa_tro_civil_war
			}
			
			random_other_country = {
				limit = {
					original_tag = USA
					has_government = trozkism
				}
				set_stability = 0.9
				add_equipment_to_stockpile = { type = infantry_equipment amount = 7000 producer = USA }
				
				add_popularity = {
					ideology = trozkism
					popularity = 0.5
				}
				add_popularity = {
					ideology = democratic
					popularity = -1
				}
			}
			
			ITA = {
				limit = {
					AND = {
						OR = {
							is_in_faction_with = SOV
							is_in_faction_with = ENG
						}
						ITA = { has_country_flag = war_success }
					}
				}
				leave_faction = { target = ITA }
			}
			
			FRA = {
				limit = {
					AND = {
						OR = {
							AND = {
								is_in_faction_with = ENG
								NOT = { has_completed_focus = FRA_go_with_britain }
							}
							is_in_faction_with = SOV
						}
						FRA = { has_country_flag = war_success }
					}
				}
				leave_faction = { target = FRA }
			}
			
			FRA = {
				limit = {
					has_country_flag = war_success
				}
				add_war_support = -50
			}
			
			ENG = {
				limit = {
					has_country_flag = war_success
				}
				add_war_support = -50
			}
		}
	}
}
