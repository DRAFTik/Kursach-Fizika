add_namespace = tr_german

# Von Mackensen dies
country_event = {
	id = tr_germany.1
	title = tr_germany.1.t
	desc = tr_germany.1.d
	picture = GFX_report_event_generic_funeral

	fire_only_once = yes
	
	trigger = {
		tag = GER
		has_government = neutrality
		NOT = { has_completed_focus = GER_revive_the_kaiserreich }
		date > 1940.1.1
		has_country_leader = { ruling_only = yes name = "August von Mackensen" }
	}

	mean_time_to_happen = {
		days = 1825	# Died in 1945
	}
	
	#nooooo sadface :(
	option = {
		name = tr_germany.1.a
		kill_country_leader = yes
		set_country_flag = GER_mackensen_dead
		create_country_leader = {
			name = "Otto Schüssler"
			desc = "POLITICS_OTTO_SCHUSSLER_DESC"
			picture = GFX_portrait_ger_otto_schussler
			expire = "1965.1.1"
			ideology = trozkism1
			traits = {
				trozkist_revolutionary
				trozkist_militarist
			}
		}
	}
}

# Ger - Offer Alliance to the Country
country_event = {
	id = tr_german.11
	title = tr_german.11.t
	desc = tr_german.11.d
	picture = GFX_report_event_degaulle_inspect_troops

	is_triggered_only = yes
	
	option = {
		name = tr_german.11.a
		ai_chance = { factor = 95 }
		GER = {
			if = {
				limit = { is_in_faction = no }
				create_faction = "GER_euro_intern"
				add_to_faction = ROOT
			}
			if = {
				limit = { 
					is_in_faction = yes
					is_faction_leader = yes
				}
				add_to_faction = ROOT
			}
			if = {
				limit = { 
					is_in_faction_with = KMX
					KMX = { is_faction_leader = yes }
				}
				KMX = {
					add_to_faction = ROOT
					country_event = { id = tr_german.14 }
				}
				ROOT = {
					add_ai_strategy = {
						type = alliance
						id = "KMX"
						value = 200
					}
				}
			}
		}
		
		ROOT = {
			add_ai_strategy = {
				type = alliance
				id = "GER"
				value = 200
			}
		}
		GER = { country_event = { id = tr_german.12 } }
	}
	
	option = {
		name = tr_german.11.b
		ai_chance = { factor = 5 }
		GER = { country_event = { id = tr_german.13 } }
	}
}
