# Greenland De Facto Independence
country_event = {
	id = wtt_denmark.2
	title = wtt_denmark.2.t
	desc = wtt_denmark.2.d
	picture = GFX_report_event_generic_sign_treaty1
	immediate = {log = "[GetDateText]: [Root.GetName]: event wtt_denmark.2"}
	
	is_triggered_only = yes
	
	option = {
		name = wtt_denmark.2.a
		release = GRO
		if = {
			limit = {
				DEN = { NOT = { has_war_with = USA } }
				USA = { is_puppet = no exists = yes } 
			}
			USA = { country_event = { id = grenland.1 days = 1 } }
		}
		else_if = {
			limit = {
				DEN = { has_war_with = USA }
				DEN = { NOT = { has_war_with = ENG } }
				ENG = { is_puppet = no exists = yes } 
			}
			ENG = { country_event = { id = grenland.1 days = 1 } }
		}
		if = {
			limit = {
				DEN = { 
					NOT = { 
						OR = { 
							is_in_faction_with = USA 
							is_in_faction_with = ENG 
						}
					} 
				}
			}
			GRO = { 
				country_event = { id = grenland.3 days = 1 } 
			}
		}
	}
}
