		name = NW_DZH.8.a

			limit = { is_subject = yes }

				set_country_flag = NW_DZH_event_11_applicant_flag

				end_puppet = ROOT
			}
		}
		every_enemy_country = {
			limit = {
				OR = {
					original_tag = FROM
					is_subject_of = FROM
					is_in_faction_with = FROM

					is_subject_of = SIK
					is_in_faction_with = SIK
				}
			}
			white_peace = ROOT
		}
		if = {
			limit = { is_in_faction = yes }
			leave_faction = yes
		}
		hidden_effect = { country_event = { id = NW_DZH.10 hours = 12 } }
	}
}
