	id = portugal.0
	title = portugal.0.t
	desc = portugal.0.d
	picture = GFX_report_event_usa_british_sailors_destroyers
	fire_only_once = yes
	
	trigger = { 
		OR = {	
			AND = {
				date > 1936.8.8
				SPR = { has_civil_war = yes }
			}
			AND = {
				date > 1936.11.1
				SPR = { has_civil_war = yes date > 1936.08.01 }
			}
		}
	}
	option = {
		name = portugal.0.a
		ai_chance = {
			factor = 1
			modifier = {
				factor = 100
				is_historical_focus_on = yes
			}
		}
		unlock_national_focus = POR_prison_kabo_verde
		if = { limit = { date > 1937.6.3 } country_event = { id = portugal.1 days = 1 } }
	}
	option = {
		name = portugal.0.b
		unlock_national_focus = POR_prison_kabo_verde
		add_stability = 0.02
		add_war_support = 0.02
		add_popularity = {
			ideology = communism
			popularity = 0.04
		}
		if = { limit = { date > 1937.6.3 } country_event = { id = portugal.2 days = 1 } }
	}
}
