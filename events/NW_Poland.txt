	id = poland.powstanie.1
	title = poland.powstanie.1.t
	desc = poland.powstanie.1.d
	
	fire_only_once = yes
	
	trigger = {
		tag = POL
		
		NOT = { POL = { has_country_flag = warszawa_porazka } } #obecnie
		
		OR = {
			AND = { #against Germany
				POL = { has_war_with = GER has_capitulated = yes }
				GER = { controls_state = 10 }
				any_country = {
					has_war_with = GER
					divisions_in_state = {
						state = 10
						size > 0
					}
				}
			}		
			AND = { #against USSR
				POL = { has_war_with = SOV has_capitulated = yes }
				SOV = { controls_state = 10 }
				any_country = {
					has_war_with = SOV
					divisions_in_state = {
						state = 10
						size > 0
					}
				}
			}		
		}
	}
	option = { # 
		name = poland.powstanie.1.a
		ai_chance = { factor = 100 }
		
		set_country_flag = POL_its_time
	}
}
