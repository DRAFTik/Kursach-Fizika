							limit = { has_character = GER_ernst_niekisch }

								set_nationality = GBF
							}
						}

								set_nationality = GBF
							}
						}
					}
					add_country_leader_role = {
						character = GER_otto_strasser

							desc="POLITICS_OTTO_STRASSER_DESC"
							ideology = strasserismus
							traits = {
								first_fighter_against_hitler
								ideologue_of_strasserism
							}
							expire = "1965.1.1"
						}
						promote_leader = yes
					}
					every_country = {
						limit = {
							original_tag = ROOT
							NOT = { tag = GBF }
						}
						declare_war_on = {
							target = GBF
							type = civil_war
						}
						add_civil_war_target = GBF
						GBF = { add_civil_war_target = PREV }
					}
					every_enemy_country = {
						every_state = {
							limit = {
								NOT = { is_fully_controlled_by = PREV }
								controller = {
									tag = PREV.PREV
								}
							}
							teleport_armies = {
								limit = {
									tag = PREV
								}
								to_state_array = controlled_states
							}
						}
					}
					delete_unit_template_and_units = {
						division_template = "SS-Sturmbrigade"
						disband = yes
					}
					delete_unit_template_and_units = {
						division_template = "Panzer-Division"
						disband = yes
					}
					delete_unit_template_and_units = {
						division_template = "Kavallerie-Brigade"
						disband = yes
					}
					delete_unit_template_and_units = {
						division_template = "Infanterie-Division (mot.)"
						disband = yes
					}
					
					load_oob = GER_German_Civil_War_strasserismus
				}
			}
