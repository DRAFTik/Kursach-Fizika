#Обсуждение проблемы Дерсима пульсар // TUR
country_event = {
	id = NWkurd.0
	is_triggered_only = yes
	hidden = yes
	
	immediate = { log = "[GetDateText]: [Root.GetName] ([Root.GetTag]): event NWkurd.0" }
	immediate = {
		if = {
			limit = {
				any_owned_state = {
					NOT = { is_core_of = ROOT }
					is_core_of = KRD
				}
			}
			country_event = { id = NWkurd.1 }
		}
	}
}

#Обсуждение проблемы Дерсима // TUR
country_event = {
	id = NWkurd.1
	title = NWkurd.1.t
	desc = NWkurd.1.desc
	picture = GFX_report_event_generic_rally_3
	is_triggered_only = yes
	
	immediate = { log = "[GetDateText]: [Root.GetName] ([Root.GetTag]): event NWkurd.1" }
	
	option = {
		name = NWkurd.1.a
		ai_chance = {
			factor = 50
		}
		add_political_power = -50
		unlock_decision_category_tooltip = KRD_problem_of_dersim_category
		set_country_flag = NWkurd1_a_flag
		set_variable = { kurdish_threat_level_var = 5 }
		if = {
			limit = { NOT = { has_active_mission = KRD_spread_of_uprising_in_dersim } }
			effect_tooltip = { activate_mission = KRD_spread_of_uprising_in_dersim }
		}
	}
	
	option = {
		name = NWkurd.1.b
		ai_chance = {
			factor = 50
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}
		}
		custom_effect_tooltip = NWkurd1_b_tt
	}
}

country_event = {
	id = NWkurd.3
	title = NWkurd.3.t
	desc = NWkurd.3.desc
	picture = GFX_report_event_generic_rally_3
	is_triggered_only = yes
	
	immediate = { log = "[GetDateText]: [Root.GetName] ([Root.GetTag]): event NWkurd.3" }
	
	option = {
		name = NWkurd.3.a
		add_political_power = -50
		add_stability = -0.05
		add_ideas = KRD_rebellion_in_dersim
		if = {
			limit = { has_country_flag = NWkurd1_a_flag }
			set_country_flag = NWkurd3_flag
			set_temp_variable = { kurdish_threat_level_var_temp = 2 }
			add_to_variable = { kurdish_threat_level_var = kurdish_threat_level_var_temp }
			clamp_variable = { var = kurdish_threat_level_var min = 0 max = 10 }
			custom_effect_tooltip = kurdish_threat_level_up_tt
			if = {
				limit = { NOT = { has_active_mission = KRD_spread_of_uprising_in_dersim } }
				effect_tooltip = { activate_mission = KRD_spread_of_uprising_in_dersim }
			}
		}
		else = { country_event = { id = NWkurd.4 days = 20 } }
	}
}
