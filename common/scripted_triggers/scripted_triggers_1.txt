		exists = yes
		NOT = {
			is_in_faction_with = ROOT
		}
		hidden_trigger = {
			OR = {
				ROOT = {
					is_ai = no
				}
				AND = {
					if = {
						limit = {
							ROOT = {
								is_major = no
								num_of_nukes < 1
							}
						}
						num_of_nukes < 1
					}
					has_volunteers_amount_from = {
						tag = ROOT
						count < 1
					}
					NOT = {
						OR = {
							strength_ratio = {
								tag = ROOT 
								ratio > 5
							}
							is_puppet_of = ROOT
							is_subject_of = ROOT
							has_war_together_with = ROOT

								OR = {
									has_guaranteed = PREV

								}
							}
						}
					}
				}
			}
		}
	}
}
