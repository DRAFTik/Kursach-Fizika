			AND = {
				has_civil_war = no

				num_divisions > 0
				has_manpower > 0
				num_of_military_factories > 0

					is_controlled_by = ROOT
				}

					limit = {
						is_in_faction = yes

					}
					is_faction_leader = yes
				}
				if = {
					limit = {
						has_war = yes
					}
					surrender_progress < 0.10
				}

					limit = {

}


target_script = {
	AND = {
		exists = yes
		NOT = {
			is_in_faction_with = ROOT
		}
		hidden_trigger = {
			OR = {
				ROOT = {
					is_ai = no
				}
				AND = {
					if = {
						limit = {
							ROOT = {
								is_major = no
								num_of_nukes < 1
							}
						}
						num_of_nukes < 1
					}
					has_volunteers_amount_from = {
						tag = ROOT
						count < 1
					}
					NOT = {
						OR = {
							strength_ratio = {
								tag = ROOT 
								ratio > 5
							}
							is_puppet_of = ROOT
							is_subject_of = ROOT
							has_war_together_with = ROOT
							gives_military_access_to = ROOT
							ROOT = {
								OR = {
									has_guaranteed = PREV
									gives_military_access_to = PREV
								}
							}
						}
					}
				}
			}
		}
	}
}
