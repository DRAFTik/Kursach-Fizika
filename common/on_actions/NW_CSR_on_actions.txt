on_actions = {
	on_startup = {
		effect = {
			### Чадонг стартпак
			if = {
				limit = {
					has_start_date < 1936.01.02
				}
				CSR = {
					country_event = { id = NWchandong.500 days = 164 }
				}
			}
		}
	}
	
	on_daily_CSR = {
		effect = {
			### УГОЛЬ
			if = {
				limit = {
					NOT = { check_variable = { modifier@coal_gain = 0 } }
				}
				add_to_variable = { coal_var_total = modifier@coal_gain }
			}
		}
	}
	
	# ROOT is subject FROM is overlord
	on_subject_autonomy_level_change = {
		effect = {
			if = {
				limit = {
					FROM = { tag = JAP }
					tag = CSR
					has_autonomy_state = autonomy_wtt_imperial_protectorate
					
				}
				hidden_effect = {
					if = {
						limit = { NOT = { has_country_flag = NWchandong_10_block_flag } }
						country_event = { id = NWchandong.10 days = 150 }
					}
					if = {
						limit = {
							has_character = MON_diluwa_khutugtu_jamsrangjab
							NOT = { has_country_flag = NWchandong_19_block_flag }
						}
						country_event = { id = NWchandong.19 days = 30 }
					}
				}
			}
		}
	}
	
	# FROM is war target
	on_declare_war = {
		effect = {
			if = {
				limit = {
					if = {
						FROM = { original_tag = MAN }
						ROOT = { original_tag = MAN }
						OR = {
							FROM = { has_government = monarchy }
							ROOT = { has_government = monarchy }
						}
						CSR = { NOT = { has_country_flag = NWchandong_14_block_flag } }
					}
				}
				hidden_effect = {
					CSR = {
						set_country_flag = NWchandong_14_block_flag
						country_event = { id = NWchandong.14 days = 1 random_days = 3 }
					}
				}
			}
		}
	}
	
	# ROOT is new controller #FROM is old controller #FROM.FROM is state ID
	on_state_control_changed = {
		effect = {
			if = {
				limit = {
					FROM.FROM = { state = 611 }
					FROM = {
						tag = CSR
						is_subject_of = JAP
						has_country_leader = { character = MEN_demchugdongrub ruling_only = yes }
						NOT = { has_country_flag = NWchandong_16_block_flag }
					}
					ROOT = { tag = MEN }
				}
				hidden_effect = {
					FROM = {
						set_country_flag = NWchandong_16_block_flag
						country_event = { id = NWchandong.16 days = 1 random_days = 3 }
					}
				}
			}
		}
	}
	
	
	on_weekly_CSR = {
		effect = {
			if = {
				limit = {
					NOT = { has_country_flag = NWchandong_15_block_flag }
					all_of_scopes = {
						array = MAN.core_states
						controller = {
							NOT = {
								tag = JAP
							}
						}
					}
				}
				set_country_flag = NWchandong_15_block_flag
				country_event = { id = NWchandong.15 days = 1 random_days = 3 }
			}
		}
	}
	
	# FROM is owner country, ROOT is the unit leader.
	on_army_leader_lost_combat = {
		effect = {
			FROM = {
				if = {
					limit = { tag = CSR NOT = { has_country_flag = CSR_leader_lost_combat_timer_flag } }
					set_country_flag = { flag = CSR_leader_lost_combat_timer_flag	days = 100 }
				}
			}
		}
	}
}
