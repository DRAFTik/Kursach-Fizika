on_actions = {
	# ROOT is capitulated country, FROM is winner
	on_capitulation = {
		effect = {
			if = {
				limit = {
					ROOT = {
						NOT = { has_country_flag = from_civil_war }
						is_major = no
					}
				}
				ROOT = { country_event = { id = NW_capitulation.1 days = 2 } }
			}
			
			if = {
				limit = {
					has_country_flag = from_civil_war
				}
				clr_country_flag = from_civil_war
			}
			
			#Второй шанс для Германии
			if = {
				limit = {
					tag = GER
					has_government = fascism
					has_civil_war = no
				}
				GER = { 
					set_country_flag = ger_second_chance_nonazi
					complete_national_focus = GER_oppose_hitler
				}
			}
			if = {
				limit = {
					tag = GER
					NOT = { has_government = fascism }
					has_civil_war = no
				}
				GER = { 
					set_country_flag = ger_second_chance_nazi
					complete_national_focus = GER_rhineland
				}
			}
			
			#Саадабад vs Союзников. Капитуляция Индии
			if = {
				limit = {
					tag = RAJ
					has_war_with = PAK
				}
				PAK = {
					puppet = RAJ
					transfer_state = 439
					transfer_state = 433
					transfer_state = 428
					transfer_state = 437
					transfer_state = 438
				}
				every_other_country = {
					limit = { has_war_with = PAK }
					every_other_country = {
						limit = {
							is_in_faction_with = PAK
							is_faction_leader = yes
						}
						white_peace = PREV
					}
				}
				if = {
					limit = {
						JAP = { is_faction_leader = yes }
						TIB = { is_in_faction_with = TUR }
					}
					every_state = {
						limit = {
							is_core_of = TIB
						}
						add_claim_by = JAP
					} 
				}
			}
			
			#Капитуляция Британии и раздача заморских террок + конференции
			if = {
				limit = {
					tag = ENG
				}
				CAN = { country_event = { id = NW_capitulation.3 days = 0 } }
				
				CAN = { leave_faction = yes }
				SAF = { leave_faction = yes }
				RAJ = { leave_faction = yes }
				MAL = { leave_faction = yes }
				AST = { leave_faction = yes }
				NZL = { leave_faction = yes }
				
				every_country = {
					limit = {
						OR = {
							CAN = { is_ai = no }
							SAF = { is_ai = no }
							RAJ = { is_ai = no }
							MAL = { is_ai = no }
							AST = { is_ai = no }
							NZL = { is_ai = no }
						}
					}
					set_autonomy = {
						target = PREV
						autonomy_state = autonomy_free
					}
				}
				
				every_other_country = {
					limit = { has_war_with = CAN }
					white_peace = CAN
				}
				every_other_country = {
					limit = { has_war_with = SAF }
					white_peace = SAF
				}
				every_other_country = {
					limit = { has_war_with = RAJ }
					white_peace = RAJ
				}
				every_other_country = {
					limit = { has_war_with = MAL }
					white_peace = MAL
				}
				every_other_country = {
					limit = { has_war_with = AST }
					white_peace = AST
				}
				every_other_country = {
					limit = { has_war_with = NZL }
					white_peace = NZL
				}
				
				if = {
					limit = {
						OR = {
							CAN = { is_ai = no }
							SAF = { is_ai = no }
							RAJ = { is_ai = no }
							MAL = { is_ai = no }
							AST = { is_ai = no }
							NZL = { is_ai = no }
						}
					}
					random_country = {
						limit = {
							OR = {
								tag = CAN
								tag = SAF
								tag = RAJ
								tag = MAL
								tag = AST
								tag = NZL
							}
							is_ai = no
						}
						country_event = { id = crush.100 days = 3 }
					}
				}
				else = {
					random_list = {
						50 = {
							CAN = { country_event = { id = crush.100 days = 3 } }
						}
						10 = {
							SAF = { country_event = { id = crush.100 days = 3 } }
						}
						10 = {
							RAJ = { country_event = { id = crush.100 days = 3 } }
						}
						10 = {
							MAL = { country_event = { id = crush.100 days = 3 } }
						}
						10 = {
							AST = { country_event = { id = crush.100 days = 3 } }
						}
						10 = {
							NZL = { country_event = { id = crush.100 days = 3 } }
						}
					}
				}
			}
			
			#Капитуляция Британии и война с турками и кипром ## Аннекс Кипра Турком
			if = {
				limit = {
					tag = ENG
					ENG = { 
						has_war_with = TUR
						has_war_with = NCP
					}
				}
				TUR = { country_event = { id = NWturkey.18 days = 1 } }
			}
			
			#Капитуляция Грека и война с турками и кипром ## Аннекс Кипра Турком
			if = {
				limit = {
					tag = GRE
					GRE = { 
						has_war_with = TUR
						has_war_with = NCP
					}
				}
				TUR = { country_event = { id = NWturkey.21 days = 1 } }
			}
		}
	}
	
	#ROOT is winner #FROM gets annexed - This will also fire on_annex
	on_civil_war_end = {
		effect = {
			FROM = {
				set_country_flag = from_civil_war
			}
		}
	}
}
