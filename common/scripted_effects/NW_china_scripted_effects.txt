set_province_border_in_array = {
	for_loop_effect = {
		start = temp_start
		end = temp_end
		value = v
		
		var:global.chinese_state^v = {
			set_temp_variable = { right = global.chinese_state^num }
			#subtract_from_temp_variable = { right = 1 }
			set_variable = { china_right_border = right }
			
			set_temp_variable = { left = temp_start }
			set_variable = { china_left_border = left }
			
			#set_variable = { china_integration_of_province_time = 120 }
		}
	}
}
