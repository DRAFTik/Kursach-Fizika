FRA_peace_category = {

	FRA_peace = {

		icon = generic_form_nation
		
		allowed = {
			original_tag = FRA
		}

		available = {
			has_capitulated = no
			16 = {
				OR = {
					is_controlled_by = GER
					is_controlled_by = ITA
				}
			}
			NOT = {
				any_state = {
					is_controlled_by = FRA
					OR = {
						is_owned_by = GER
						is_owned_by = ITA
					}
				}
			}
		}
		
		visible = {
			has_country_flag = FRA_peace_flag
			has_government = neutrality
			OR = {
				AND = {
					has_war_with = GER
					NOT = { has_war_with = ITA }
				}
				AND = {
					has_war_with = ITA
					NOT = { has_war_with = GER }
				}
				AND = {
					has_war_with = GER
					has_war_with = ITA
					ITA = { is_in_faction_with = GER }
				}
			}
			FROM = {
				has_event_target = france_occupy_leader
			}
		}
		
		target_trigger = {
			FROM = {
				has_event_target = france_occupy_leader
			}
		}
		
		cost = 20
		fire_only_once = yes

		complete_effect = {
			event_target:france_occupy_leader = { country_event = { id = NWfrance.102 days = 1 } }
			clr_country_flag = FRA_peace_flag
		}

		ai_will_do = {
			factor = 100
		}
	}
}
