TUR_restore_caliphate_strategy = {
	allowed = {
		OR = {
			original_tag = TUR
			original_tag = TRK
		}
	}
	enable = {
		focus_progress = { focus = TUR_triumph_of_forces_of_order progress > 0 }
		country_exists = SAU
		SAU = { is_subject = no }
	}
	
	abort = {
		OR = {
			NOT = { country_exists = SAU }
			SAU = { is_subject = yes }
			has_completed_focus = TUR_restore_caliphate
			controls_state = 679
		}
	}

	ai_strategy = {
		type = antagonize
		id = "SAU"
		value = 200
	}
	ai_strategy = {
		type = conquer
		id = "SAU"
		value = 200
	}
	ai_strategy = {
		type = declare_war
		id = "SAU"
		value = 100
	}
}

if_you_between_TUR_and_SAU_you_die = {
	reversed = yes
	enable = {
		TUR = {
			focus_progress = { focus = TUR_triumph_of_forces_of_order progress > 0 }
			NOT = {
				is_neighbor_of = SAU
			}
		}
		country_exists = SAU
		SAU = { is_subject = no }
		is_neighbor_of = SAU
		is_neighbor_of = TUR
		any_country = {
			is_neighbor_of = SAU
			is_neighbor_of = TUR
			NOT = { has_war_with = TUR }
		}
	}
	
	abort = {
		OR = {
			NOT = { country_exists = SAU }
			SAU = { is_subject = yes }
			TUR = {
				OR = {
					has_completed_focus = TUR_restore_caliphate
					is_neighbor_of = SAU
				}
			}
			any_country = {
				is_neighbor_of = SAU
				is_neighbor_of = TUR
				has_war_with = TUR
			}
		}
	}

	ai_strategy = {
		type = antagonize
		id = "TUR"
		value = 400
	}
	ai_strategy = {
		type = conquer
		id = "TUR"
		value = 400
	}
	ai_strategy = {
		type = declare_war
		id = "TUR"
		value = 200
	}
}
