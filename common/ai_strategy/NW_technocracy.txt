crazy_technocracy_against_GER = {
	allowed = {
		NOT = { tag = GER }
	}
	enable = {
		num_divisions > 0
		has_completed_focus = concept_of_technate
		has_government = technocracy
		is_neighbor_of = GER
		NOT = {
			tag = GER
			is_in_faction_with = GER
			is_subject_of = GER
		}
		GER = { NOT = { is_subject_of = ROOT } }
	}
	abort_when_not_enabled = yes
	
	ai_strategy = { type = antagonize id = "GER" value = 500 }
	ai_strategy = { type = conquer id = "GER" value = 500 }
	ai_strategy = { type = prepare_for_war id = "GER" value = 500 }
	ai_strategy = { type = declare_war id = "GER" value = 500 }
}
