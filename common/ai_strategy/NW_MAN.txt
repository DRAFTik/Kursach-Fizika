	allowed = {
		tag = MAN
	}
	enable = {
		tag = MAN
		has_war_with = CHI
		country_exists = CHI
	}

		NOT = {
			country_exists = CHI
			has_war_with = CHI
		}
	}
	
	ai_strategy = {
		type = antagonize
		id = "CHI"			
		value = 200
	}

		type = invade
		id = "CHI"			
		value = 100
	}
}
